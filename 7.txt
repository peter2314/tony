import java.util.*;

public class FCFS_RR {
    static class Process {
        int pid, arrivalTime, burstTime, remaining, completion, waiting, turnaround;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of processes: ");
        int n = sc.nextInt();
        Process[] p = new Process[n];
        for (int i = 0; i < n; i++) {
            p[i] = new Process();
            p[i].pid = i + 1;
            System.out.println("\nEnter details for Process " + (i + 1));
            System.out.print("Arrival Time: ");
            p[i].arrivalTime = sc.nextInt();
            System.out.print("Burst Time: ");
            p[i].burstTime = sc.nextInt();
            p[i].remaining = p[i].burstTime;
        }

        System.out.print("Enter Time Quantum for Round Robin: ");
        int tq = sc.nextInt();

        simulateFCFS(p);
        simulateRR(p, tq);
        sc.close();
    }

    static void simulateFCFS(Process[] input) {
        Process[] p = Arrays.stream(input).map(FCFS_RR::copy).toArray(Process[]::new);
        Arrays.sort(p, Comparator.comparingInt(a -> a.arrivalTime));
        int time = 0;
        for (Process pr : p) {
            if (time < pr.arrivalTime) time = pr.arrivalTime;
            pr.completion = time + pr.burstTime;
            pr.turnaround = pr.completion - pr.arrivalTime;
            pr.waiting = pr.turnaround - pr.burstTime;
            time = pr.completion;
        }
        System.out.println("\n=== FCFS Scheduling ===");
        printTable(p);
    }

    static void simulateRR(Process[] input, int tq) {
        Process[] p = Arrays.stream(input).map(FCFS_RR::copy).toArray(Process[]::new);
        Queue<Process> q = new LinkedList<>();
        int time = 0, completed = 0;
        Arrays.sort(p, Comparator.comparingInt(a -> a.arrivalTime));

        q.add(p[0]);
        boolean[] inQueue = new boolean[p.length];
        inQueue[0] = true;

        while (completed != p.length) {
            Process curr = q.poll();
            if (curr == null) {
                time++;
                continue;
            }
            if (curr.remaining > tq) {
                curr.remaining -= tq;
                time += tq;
            } else {
                time += curr.remaining;
                curr.remaining = 0;
                curr.completion = time;
                curr.turnaround = curr.completion - curr.arrivalTime;
                curr.waiting = curr.turnaround - curr.burstTime;
                completed++;
            }

            for (Process pr : p) {
                if (!inQueue[pr.pid - 1] && pr.arrivalTime <= time && pr.remaining > 0) {
                    q.add(pr);
                    inQueue[pr.pid - 1] = true;
                }
            }
            if (curr.remaining > 0) q.add(curr);
        }

        System.out.println("\n=== Round Robin Scheduling ===");
        printTable(p);
    }

    static Process copy(Process x) {
        Process p = new Process();
        p.pid = x.pid; p.arrivalTime = x.arrivalTime; p.burstTime = x.burstTime; p.remaining = x.remaining;
        return p;
    }

    static void printTable(Process[] p) {
        System.out.println("PID\tAT\tBT\tCT\tTAT\tWT");
        double totalTAT = 0, totalWT = 0;
        for (Process pr : p) {
            totalTAT += pr.turnaround;
            totalWT += pr.waiting;
            System.out.printf("P%d\t%d\t%d\t%d\t%d\t%d\n", pr.pid, pr.arrivalTime, pr.burstTime,
                    pr.completion, pr.turnaround, pr.waiting);
        }
        System.out.printf("Average TAT = %.2f, Average WT = %.2f\n", totalTAT / p.length, totalWT / p.length);
    }
}
